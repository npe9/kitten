299 
300 !-----------------------------------------------------------------------
301 !
302 ! Setup the material according to mat_opt.
303 !
304 ! There are only two materials max; one per cell. mat_opt defines the
305 ! material layout and has a similar meaning for 1-D, 2-D, 3-D problems.
306 !
307 ! 0 = Homogeneous (mat1) problem, regardless of dimension
308 ! 1 = Center. mat1 is the base. 1-D: 25%mat1/50%mat2/25%mat1
309 ! 2/D: same as 1-D but in two dimensions, so mat2 region is a square
310 ! 3/D: same as 1-D but in three dimensions, so mat2 is a cube
311 ! 2 = Corner. Same concept as 1, but move slab/square/cube to origin
312 !
313 ! Return starting/ending indices for printint an echo.
314 !
315 !-----------------------------------------------------------------------

323  INTEGER(i_knd) :: i, j, k, jj, kk
324 !_______________________________________________________________________

327 ! the rest of the layout.

333  j1 = 1; j2 = 1

336  i1 = nx/4 + 1

345  END IF

347  i2 = nx/2
348  IF ( ndimen > 1 ) THEN
349  j2 = ny_gl/2
350  IF ( ndimen > 2 ) k2 = nz_gl/2
351  END IF
352  END IF
353

355  DO k = k1, k2
356  IF ( klb<=k .AND. k<=kub ) THEN
357  kk = mod( k-1, nz ) + 1
358  DO j = j1, j2
359  IF ( jlb<=j .AND. j<=jub ) THEN
360  jj = mod( j-1, ny ) + 1
361  DO i = i1, i2
362  mat(i,jj,kk) = 2
363  END DO

377 !-----------------------------------------------------------------------
378 !
379 ! Setup the material according to src_opt.
380 !
381 ! Source is either on at strength 1.0 or off, per cell. src_opt defines
382 ! the layout and has a similar meaning for 1-D, 2-D, 3-D problems.
383 !
384 ! 0 = Source everywhere, regardless of dimension
385 ! 1 = Source occupies the slab/square/cube center of the problem

387 ! 2/D: same as 1-D but in two dimensions
388 ! 3/D: same as 1-D but in three dimensions
389 ! 2 = Corner. Same concept as 1, but move slab/square/cube to origin
390 ! 3 = MMS option, f = sin(pi*x/lx)
391 ! All options isotropic source
392 !

394 !

471  SUBROUTINE setup_data
472 
473 !-----------------------------------------------------------------------
474 !
475 ! Setup the data arrays according to simple rules/functions to scale
476 ! to arbitrary number of groups, moments. Fixed at a library for two
477 ! materials only. Fixed group-to-group coupling properties.
478 !
479 !-----------------------------------------------------------------------
480 !_______________________________________________________________________
481 !
482 ! Local variables
483 !_______________________________________________________________________
484 
485  INTEGER(i_knd) :: g, n
486 
487  REAL(r_knd) :: t
488 !_______________________________________________________________________
489 !
490 ! Set the group 1 data for material 1 and material 2 (if present).
491 !_______________________________________________________________________
492 
493  sigt(1,1) = one
494  siga(1,1) = half
495  sigs(1,1) = half
496 
497  IF ( nmat == 2 ) THEN
498  sigt(2,1) = two
499  siga(2,1) = 0.8_r_knd
500  sigs(2,1) = 1.2_r_knd
501  END IF
502 !_______________________________________________________________________
503 !
504 ! Scale these values for increasing number of groups by adding 0.01
505 ! to sigt for each additional group. Split that 0.01 evenly among
506 ! siga and sigs.
507 !_______________________________________________________________________
508 
509  DO g = 2, ng
510  sigt(:,g) = sigt(:,g-1) + 0.01_r_knd
511  siga(:,g) = siga(:,g-1) + 0.005_r_knd
512  sigs(:,g) = sigs(:,g-1) + 0.005_r_knd
513  END DO
514 !_______________________________________________________________________
515 !
516 ! For material 1, upscattering from a given group to all above groups
517 ! is 10% of total scattering. 20% is in group. 70% is down-scattering.
518 ! For group 1/ng, no up/down-scattering, so self-scattering takes the
519 ! remaining fraction.
520 !_______________________________________________________________________
521 
522  DO g = 1, ng
523 
524  IF ( ng == 1 ) THEN
525  slgg(1,1,1,1) = sigs(1,g)
526  exit
527  END IF
528 
529  slgg(1,1,g,g) = 0.2_r_knd*sigs(1,g)
530 
531  IF ( g > 1 ) THEN
532  t = one / REAL( g-1, r_knd )
533  slgg(1,1,g,1:(g-1)) = 0.1_r_knd*sigs(1,g)*t
534  ELSE
535  slgg(1,1,1,1) = 0.3_r_knd*sigs(1,1)
536  END IF
537 
538  IF ( g < ng ) THEN
539  t = one / REAL( ng-g, r_knd )
540  slgg(1,1,g,(g+1):ng) = 0.7_r_knd*sigs(1,g)*t
541  ELSE
542  slgg(1,1,ng,ng) = 0.9_r_knd*sigs(1,ng)
543  END IF
544 
545  END DO
546 
547  IF ( nmat == 2 ) THEN
548 !_______________________________________________________________________
549 !
550 ! Repeat for material 2. Up-scattering is 10%. In-scattering is 50%.
551 ! Down-scattering is 40%.
552 !_______________________________________________________________________
553 
554  DO g = 1, ng
555 
556  IF ( ng == 1 ) THEN
557  slgg(2,1,1,1) = sigs(2,g)
558  exit
559  END IF
560 
561  slgg(2,1,g,g) = 0.5_r_knd*sigs(2,g)
562 
563  IF ( g > 1 ) THEN
564  t = one / REAL( g-1, r_knd )
565  slgg(2,1,g,1:(g-1)) = 0.1_r_knd*sigs(2,g)*t
566  ELSE
567  slgg(2,1,1,1) = 0.6_r_knd*sigs(2,1)
568  END IF
569 
570  IF ( g < ng ) THEN
571  t = one / REAL( ng-g, r_knd )
572  slgg(2,1,g,(g+1):ng) = 0.4_r_knd*sigs(2,g)*t
573  ELSE
574  slgg(2,1,ng,ng) = 0.9_r_knd*sigs(2,ng)
575  END IF
576 
577  END DO
578 
579  END IF
580 !_______________________________________________________________________
581 !
582 ! Set group-to-group scattering moments. Allowed up 4 moments. For
583 ! material 1, start with the base above. Divide by 10 to get next
584 ! moment's data. Repeat that by 2 continuously, up to nmom or 4.
585 !_______________________________________________________________________
586 
587  IF ( nmom > 1 ) THEN
588  slgg(1,2,:,:) = 0.1_r_knd*slgg(1,1,:,:)
589  DO n = 3, nmom
590  slgg(1,n,:,:) = half*slgg(1,n-1,:,:)
591  END DO
592  END IF
593 !_______________________________________________________________________
594 !
595 ! Similar procedure for material 2, but first multiply by 0.8. Then
596 ! reduce the data magnitudes by 0.6 for each successive moment.
597 !_______________________________________________________________________
598 
599  IF ( nmat==2 .AND. nmom>1 ) THEN
600  slgg(2,2,:,:) = 0.8_r_knd*slgg(2,1,:,:)
601  DO n = 3, nmom
602  slgg(2,n,:,:) = 0.6_r_knd*slgg(2,n-1,:,:)
603  END DO
604  END IF
605 !_______________________________________________________________________
606 !_______________________________________________________________________
607 
608  END SUBROUTINE setup_data
